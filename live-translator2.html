let recognition = null, isListening = false, isTranslating = false, translateTimer = null, finalText = '', lastResult = '', lastLang = 'ja'; const orb = document.getElementById('orb'), statusEl = document.getElementById('status'), sourceBox = document.getElementById('sourceBox'), resultBox = document.getElementById('resultBox'), speakBtn = document.getElementById('speakBtn'); function toggleListen() { isListening ? stopListen() : startListen(); } function startListen() { const SR = window.SpeechRecognition || window.webkitSpeechRecognition; if(!SR) { setStatus('❌ 請用 Chrome', ''); return; } finalText = ''; recognition = new SR(); recognition.lang = 'zh-TW'; recognition.continuous = true; recognition.interimResults = true; recognition.onstart = () => { isListening = true; orb.classList.add('listening'); orb.textContent = '⏹'; setStatus('🎤 聆聽中…', 'listening'); }; recognition.onresult = (e) => { let interim = ''; for(let i = e.resultIndex; i < e.results.length; i++) { if(e.results[i].isFinal) finalText += e.results[i][0].transcript; else interim += e.results[i][0].transcript; } sourceBox.innerHTML = finalText + (interim ? `${interim}` : ''); clearTimeout(translateTimer); if(finalText.trim()) { translateTimer = setTimeout(() => { doTranslate(finalText.trim()); finalText = ''; }, 900); } }; recognition.onerror = (e) => { if(e.error === 'no-speech' || e.error === 'aborted') return; setStatus('⚠️ ' + e.error, ''); }; recognition.onend = () => { if(isListening && !isTranslating) { setTimeout(() => { if(isListening && !isTranslating) { try { recognition.start(); } catch(err) {} } }, 300); } }; try { recognition.start(); } catch(err) {} } function stopListen() { isListening = false; isTranslating = false; if(recognition) recognition.stop(); orb.classList.remove('listening'); orb.classList.remove('thinking'); orb.textContent = '🎤'; setStatus('已停止', ''); } async function doTranslate(text) { isTranslating = true; if(recognition) recognition.abort(); orb.classList.remove('listening'); orb.classList.add('thinking'); orb.textContent = '⏳'; setStatus('翻譯中…', 'thinking'); const isJa = /[\u3040-\u30ff]/.test(text); const sl = isJa ? 'ja' : 'zh-TW'; const tl = isJa ? 'zh-TW' : 'ja'; lastLang = tl; try { const res = await fetch(`https://api.mymemory.translated.net/get?q=${encodeURIComponent(text)}&langpair=${sl}|${tl}`); const data = await res.json(); const t = data.responseData?.translatedText || ''; if(!t) throw new Error(); lastResult = t; resultBox.textContent = t; speakBtn.disabled = false; speak(t, tl === 'ja' ? 'ja-JP' : 'zh-TW'); } catch { setStatus('❌ 翻譯失敗', ''); resumeListening(); } } function speak(text, lang) { const u = new SpeechSynthesisUtterance(text); u.lang = lang; u.rate = 0.9; u.onend = resumeListening; u.onerror = resumeListening; window.speechSynthesis.cancel(); window.speechSynthesis.speak(u); } function resumeListening() { isTranslating = false; if(isListening) { orb.classList.remove('thinking'); orb.classList.add('listening'); orb.textContent = '⏹'; setStatus('🎤 聆聽中…', 'listening'); try { recognition.start(); } catch(err) {} } } function speakAgain() { if(lastResult) { isTranslating = true; if(recognition) recognition.abort(); speak(lastResult, lastLang === 'ja' ? 'ja-JP' : 'zh-TW'); } }
